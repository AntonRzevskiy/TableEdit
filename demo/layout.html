Установка

Новинка Внимание Осуществляется в течении 1 минуты.

Плагин для создания редактируемой таблицы из массива и не только. Вы можете легко добавлять и удалять строки, ячейки. Плагин содержит достаточно возможностей и функций обратного вызова для быстрой настройки вашей задачи.

jQuery(document).ready(function($){

    $('#from-textarea').tableEdid({
        outputMethod: 'append',
        outputConteiner: '#table-container',
        compileTableAfter: function() {
            this.$table.addClass('tableEdit');
            return true;
        },
        eventsBindAfter: function() {
            $('#save-table').on('click',this,function(e){
                $('#from-textarea').val( JSON.stringify(e.data.dataTableArray) );
            });
            return true;
        },
        stubElements: function() {
            $(this).css("width","20%");
            return '<a class="addCol" href="javascript://" role="button"><span class="glyphicon glyphicon-plus"></span></a>';
        }
    });

});
